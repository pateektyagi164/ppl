%union {
    int num;
    char id;
}

%token <num> NUMBER
%token <id> ID

%left '+' '-'
%left '*' '/'
%right '^'
%right UMINUS

%%
input:
      /* empty */
    | input line
;

line:
    expr '\n' { printf("\n"); }
;

expr:
      expr '+' expr { printf("+"); }
    | expr '-' expr { printf("-"); }
    | expr '*' expr { printf("*"); }
    | expr '/' expr { printf("/"); }
    | expr '^' expr { printf("^"); }
    | '(' expr ')'
    | '-' expr %prec UMINUS {printf("-");}
    | NUMBER        { printf("%d", $1); }
    | ID            { printf("%c", $1); }
;

printf("error : %s\n",s);
