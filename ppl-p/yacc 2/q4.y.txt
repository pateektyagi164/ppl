%{
#include <stdio.h>
#include <stdlib.h>

int yylex();
void yyerror(const char *s);
%}

%union {
    int type;  
}

%token <type> NUMBER
%token ID

%type <type> expr
%left '+' '-'
%left '*' '/'
%right UMINUS

%%
stmt:
    ID '=' expr ';' {
        if($3 == 0)
            printf("Valid: result is INT\n");
        else
            printf("Valid: result is FLOAT\n");
    }
;

expr:
      expr '+' expr {
            $$ = ($1 == 1 || $3 == 1) ? 1 : 0;
      }
    | expr '-' expr {
            $$ = ($1 == 1 || $3 == 1) ? 1 : 0;
      }
    | expr '*' expr {
            $$ = ($1 == 1 || $3 == 1) ? 1 : 0;
      }
    | expr '/' expr {
            $$ = 1;   // division always float
      }
    | '-' expr %prec UMINUS {$$=($2==1)?1:0;}
    | NUMBER { $$ = $1; }
;

%%
void yyerror(const char *s){
    printf("Error: %s\n", s);
}

int main(){
    printf("Enter expression:\n");
    yyparse();
    return 0;
}
