[0-9]+        { yylval.type = 0; return NUMBER; }        // int
[0-9]+\.[0-9]+ { yylval.type = 1; return NUMBER; }      // float

[a-zA-Z_][a-zA-Z0-9_]*     { return ID; }

[ \t\n]       {}

"+"           { return '+'; }
"-"           { return '-'; }
"*"           { return '*'; }
"/"           { return '/'; }
"="           { return '='; }
";"           { return ';'; }

. { printf("Invalid input\n");exit(1); }
