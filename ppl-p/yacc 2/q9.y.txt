%union{
double dval;
}
%token <dval> NUMBER
%token NEWLINE
%type <dval> expr
%left '+' '-'
%left '*' '/'
%right UMINUS

%%
input: |input line;
line: expr NEWLINE {printf("\n%.2lf",$1);};
expr: expr '+' expr {$$=$1+$3;}
     |expr '-' expr {$$=$1-$3;}
     |expr '/'expr {if($3==0){
                    printf("division by zero is detected \n");YYABORT;
                    }else{
                 $$=$1/$3;
              };
             }
     |expr '*'expr {$$=$1*$3;}
     |'(' expr ')' {$$=$2;}
     | NUMBER {$$=$1;}
     |'-'expr %prec UMINUS {$$=-$2;};
%%
void yyerror(const char *s){
   printf("error:%s\n",s);
}
