%{
#include <stdio.h>
#include <stdlib.h>

void yyerror(const char *s);
int yylex();
%}

%token ID NUMBER

%%
func:
    ID '(' args ')' ';' { printf("Valid function call\n"); }
;

args:
      /* empty */                // for f()
    | arg_list
;

arg_list:
      arg
    | arg_list ',' arg
;

arg:
      ID
    | NUMBER
;

%%

void yyerror(const char *s) {
    printf("Error: %s\n", s);
}

int main() {
    printf("Function call parser\n");
    yyparse();
    return 0;
}
