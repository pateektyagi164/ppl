%{
#include<stdio.h>
#include<stdlib.h>

int yylex();
void yyerror(const char *s);
%}

%token SWITCH CASE DEFAULT BREAK NUMBER ID

%%
program:
    switch_stmt { printf("Valid switch-case\n"); }
;

switch_stmt:
    SWITCH '(' ID ')' '{' case_list default_opt '}'
;

case_list:
      case_list case_stmt
    | case_stmt
;

case_stmt:
    CASE NUMBER ':' stmt_list BREAK ';'
;

default_opt:
      DEFAULT ':' stmt_list
    | /* empty */
;

stmt_list:
      stmt
    | stmt_list stmt
;

stmt:
    ID '=' NUMBER ';'
;

%%

void yyerror(const char *s){
    printf("Invalid switch-case\n");
}

int main(){
    printf("Enter switch-case:\n");

    if(yyparse() == 0){
        printf("PARSED SUCCESSFULLY\n");
    }

    return 0;
}
