%type  expr
%left '+' '-'
%left '*' '/'
%right UMINUS

     |expr '-' expr 
     |expr '/'expr 
     |expr '*'expr 
     |'(' expr ')' 
     | NUMBER 
     |'-'expr %prec UMINUS ;
