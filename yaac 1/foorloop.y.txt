%{
#include<stdio.h>
#include<stdlib.h>
#include<stdbool.h>
void yyerror(const char *s);
int yylex();
%}
%token FOR IF ELSE NUMBER ID GT LT EQQ EQ LTE GTE 
%left '+' '-'
%left '*' '/'
%right UMINUS
%%
program: for_st | program for_st ;
for_st: FOR '('ID EQ exp ';' condition ';' ID EQ exp ')' '{' statement_st '}'; 
statement_st: statement|statement_st statement;
statement: IF '(' condition ')' '{' statement_st '}'|
          IF '(' condition ')' '{' statement_st '}' ELSE '{' statement_st '}'|
          ID EQ exp ';';
exp: NUMBER | exp '-' exp | exp '+' exp | exp '*' exp | exp '/' exp | '-' exp %prec UMINUS|ID;

condition: exp GT exp| exp LT exp | exp EQQ exp | exp LTE exp | exp GTE exp;
%%
void yyerror(const char *s){
printf("\n error :%s",s);
}
int main(){
printf("enter the for loop \n");

        printf("PARSED SUCCESSFULLY\n");
    };

}
